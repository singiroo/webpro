1.charAt/charCodeAt

str = "가나다/abc/Abc/사자/호랑이/고양이/치타/표범/퓨마"
charAt(index) : index위치의 문자를 리턴
charCodeAt(index) : index위치의 unicode값을 리턴.



2.indexOf/lastIndexOf

str = "가나다/abc/Abc/사자/호랑이/고양이/치타/표범/퓨마/사자/가나다"
indexOf("문자열") : 문자열의 위치를 검색
lastIndexOf("문자열") : 문자열의 위치를 검색, 단 끝에서부터 시작. 인덱스는 그대로



3.concat/trim

str = "가나다/abc/Abc/사자/호랑이/고양이/치타/표범/퓨마/사자/가나다"
str2 = "사과/복숭아/자두/참외/수박/바나나/귤"
concat() : 문자열 합치기
str4 = "      사과 바나나 귤 자두 참외 멜론                 "
trim() : 앞 뒤 공백 제거



4.substring/slice/substr

str = "가나다/abc/Abc/사자/호랑이/고양이/치타/표범/퓨마/사자/가나다"
str2 = "사과/복숭아/자두/참외/수박/바나나/귤"
substring(start, end) : start ~ end-1까지의 문자열을 추출
slice(start, end) : start ~ end-1까지의 문자열을 추출
substr(start, length) : start부터 length 갯수 만큼 추출



5.split

str = "가나다/abc/Abc/사자/호랑이/고양이/치타/표범/퓨마/사자/가나다"
split(구분자, 갯수) : 구분자로 문자를 구분해서 배열로 리턴.



6.replace

str = "가나다/abc/Abc/사자/호랑이/고양이/치타/표범/퓨마/사자/가나다"
replace(old, new) : old의 값을 new의 값으로 변경.
replace(//g) /와 /사이에 정규 표현식을 사용할 수 있다, g : global : 전체 영역?
